──────────────────────────────────────────────────────────────
 DOCKER İLE ÇALIŞTIRMA (ÖNERİLEN - EN KOLAY)
──────────────────────────────────────────────────────────────

git clone https://github.com/DevKursat/linkedinAgent.git

cd linkedinAgent

copy .env.example .env

REM ⚠️ ŞİMDİ .env DOSYASINI DÜZENLE:
REM - LINKEDIN_CLIENT_ID, LINKEDIN_CLIENT_SECRET, GEMINI_API_KEY ekle
REM - LinkedIn Developer Portal: https://www.linkedin.com/developers/apps
REM - Google Gemini API: https://makersuite.google.com/app/apikey

docker compose up -d --build

start http://localhost:5000

REM ✅ Tarayıcıda "Login with LinkedIn" tıkla


──────────────────────────────────────────────────────────────
 PYTHON İLE ÇALIŞTIRMA (DOCKER YOKSA)
──────────────────────────────────────────────────────────────

git clone https://github.com/DevKursat/linkedinAgent.git

cd linkedinAgent

python -m venv .venv

.venv\Scripts\activate

pip install -r requirements.txt

copy .env.example .env

REM ⚠️ ŞİMDİ .env DOSYASINI DÜZENLE:
REM - LINKEDIN_CLIENT_ID, LINKEDIN_CLIENT_SECRET, GEMINI_API_KEY ekle
REM - LINKEDIN_REDIRECT_URI=http://127.0.0.1:8000/callback yap

REM TERMINAL 1 - Web Server başlat:
uvicorn src.main:app --reload --host 127.0.0.1 --port 8000

REM YENİ TERMINAL 2 - Worker başlat:
.venv\Scripts\activate
python -m src.worker

REM Tarayıcıda aç:
start http://127.0.0.1:8000


──────────────────────────────────────────────────────────────
 TEST YAPMA (GÖNDERİ YAPMADAN ÖNCE)
──────────────────────────────────────────────────────────────

REM Docker ile:
docker compose exec worker python -c "from src.worker import trigger_post_creation; trigger_post_creation()"

REM Python ile:
.venv\Scripts\activate
python -c "from src.worker import trigger_post_creation; trigger_post_creation()"

REM ✅ "[DRY_RUN] Would post:" mesajını göreceksin


──────────────────────────────────────────────────────────────
 CANLIYA ALMA (TEST SONRASI)
──────────────────────────────────────────────────────────────

REM .env dosyasında DRY_RUN=false yap

REM Docker ile yeniden başlat:
docker compose restart worker

REM Python ile: Worker terminalini kapat (Ctrl+C) ve tekrar başlat:
python -m src.worker
